Início
    Para i de 0 até 100, com passo 2 faça
        Imprimir i
    Fim Para

(java)
package pares;

public class Pares {

    public static void main(String[] args) {
        for (int i =0; i <=100; i +=2) {
            System.out.println(i);
        }
    } 
}
